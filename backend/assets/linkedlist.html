Understanding Linked Lists in C++

What is a Linked List?

A linked list is a linear data structure where each element is a separate object, called a node. Each node contains two parts:

  1. Data: The value or information of the node.
  2. Next: A pointer or reference to the next node in the sequence.

Unlike arrays, linked lists do not have a fixed size and can grow or shrink dynamically. This makes them more flexible for certain applications.

Types of Linked Lists

  • Singly Linked List: Each node points to the next node.
  • Doubly Linked List: Each node points to both the next and previous nodes.
  • Circular Linked List: The last node points back to the first node, forming a circle.

Implementing a Linked List

Defining a Node

class Node {
public:
    int data;
    Node *next;

    Node() {}
    ~Node() {}
    Node(int d) : data(d), next(nullptr) {}
    Node(int d, Node *next) : data(d), next(next) {}
};

Defining a Linked List


class LinkedList {
private:
    Node *HEAD;
    Node *TAIL;

public:
    LinkedList::LinkedList() {
        this->HEAD = nullptr;
        this->TAIL = nullptr;
    }
    LinkedList::~LinkedList() {
        this->FreeMemory();
    }
    void addToHead(int data);
    void addToTail(int data);
    bool removeFromHead();
    bool removeFromTail();
    Node *searchsmt(int target);
    void traverse();
    bool remove(int data);
    void add(Node *pred, int data);
    bool isEmpty();
};

Different Operations

Add to Head

void LinkedList::addToHead(int data) {
    Node *temp = new Node(data, this->HEAD);
    this->HEAD = temp;
    if (this->TAIL == nullptr) {
        this->TAIL = this->HEAD;
    }
}

Add to Tail

void LinkedList::addToTail(int data) {
    if (this->isEmpty()) {
        Node *temp = new Node(data, nullptr);
        this->TAIL = temp;
        this->HEAD = temp;
    } else {
        Node *temp = new Node(data, nullptr);
        this->TAIL->next = temp;
        this->TAIL = temp;
    }
}

Remove from Head

bool LinkedList::removeFromHead() {
    if (!this->isEmpty()) {
        Node *nodeToDelete = this->HEAD;
        this->HEAD = nodeToDelete->next;
        delete nodeToDelete;
        return true;
    } else {
        return false;
    }
}

Remove from Tail

bool LinkedList::removeFromTail() {
    if (this->isEmpty()) {
        std::cout << "Nothing is there";
        return false;
    } else {
        Node *prev = this->HEAD;
        while (prev->next->next != nullptr) {
            prev = prev->next;
        }
        delete prev->next;
        prev->next = nullptr;
        TAIL = prev;
        return true;
    }

    return false;
}

Remove by Value

bool LinkedList::remove(int target) {
    if (this->isEmpty()) {
        std::cout << "Nothing to remove";
        return false;
    } else if (target == HEAD->data || target == TAIL->data) {
        std::cout << "User remove from head or tail";
        return false;
    } else {
        Node *previous = HEAD;
        Node *current = HEAD->next;
        while (current != nullptr) {
            if (current->data == target) {
                break;
            }
            previous = current;
            current = current->next;
        }
        if (current == nullptr) {
            std::cout << "Not found";
            return false;
        } else {
            previous->next = current->next;
            if (current == TAIL) {
                TAIL = previous;
            }
            delete current;
            return true;
        }
    }
    return false;
}
