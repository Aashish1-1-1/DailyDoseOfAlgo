Circular Queue

What is a Circular Queue?

A Circular Queue is an extended version of a normal queue where the last element of the queue is connected to the first element of the queue forming a circle

Prerequisite

ArrayQueue

Problem with regular Queue?

The queue gradually shifts to the right.Example: suppose maxQueueSize = 4,and operations performed are enqueue(2), enqueue(5), dequeue(),enqueue(4), dequeue(), enqueue(3).

Implementing a Circular Queue

Defining a Queue Interface

class Queue{
	public:
		Queue() {};
		~Queue() {};
		virtual bool isEmpty()=0;
		virtual void enqueue(int max)=0;
		virtual int dequeue()=0;
		virtual int front()=0;
		virtual int back()=0;
};

Defining a Circular Queue


#include "queue.h"

class CircularQueue : public Queue{
	public:
		CircularQueue(int max){
	        	this->arr = new int[max];
	        	this->maxsize=max;
	        	this->frontindex=0;
	        	this->backindex=0;
		};
		~CircularQueue(){
			delete [] arr;
		};
		
		 bool isEmpty();
		 bool isFull();
		 void enqueue(int data);
		 int dequeue();
		 int front();
		 int back();
	private:
		 int *arr;
		 int frontindex;
		 int backindex;
		 int maxsize;
};

Different Operations

IsEmpty

bool CircularQueue::isEmpty(){
	if(frontindex==backindex){
		return true;
	}
	 return false;
}

IsFull

bool CircularQueue::isFull(){
	if(frontindex==(backindex+1)%this->maxsize){
		return true;
	}
	return false;
}

Enqueue

void CircularQueue::enqueue(int data){
	if(this->isFull()){
		std::cout<<"Overflow";
	}
	else{
		*(arr+backindex)=data;
		backindex=(backindex+1)%this->maxsize;
	}
}

Dequeue

int CircularQueue::dequeue(){
	if(this->isEmpty()){
		std::cout<<"Underflow";
		return -1;
	}
	int toreturn=*(arr+frontindex);
	frontindex=(frontindex+1)%this->maxsize;
	return toreturn;
}

Front

int CircularQueue::front(){
	return *(arr+frontindex);
}

Back

int CircularQueue::back(){
	return *(arr+backindex);
}
